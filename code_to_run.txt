# Environment setup (main pipeline)
Create and install the main environment used for:
QC, dataset generation, feature extraction, training, evaluation.

python -m venv .venv
source .venv/bin/activate
python -m pip install -U pip
python -m pip install -r requirements.txt

# Forward-model sanity check (QC)
This produces the forward‑model QC figures under plots/qc_forward/.

source .venv/bin/activate
python -m eval.qc_forward --n 50 --seed 0 --out plots/qc_forward


# Generate synthetic CMC dataset (warmup + regime rejection)
This creates the H5 dataset used by everything downstream.

source .venv/bin/activate
python -m data.generate_dataset --n 10000 --seed 0 --out data/synthetic_cmc_dataset.h5 --overwrite


# Extract token features + fixed splits + token QC

Writes to data_out/.
source .venv/bin/activate
python -m data.prepare_training_data --in data/synthetic_cmc_dataset.h5 --out-dir data_out --overwrite

Expected outputs:
data_out/features.npy
data_out/features_erp.npy
data_out/features_tfr.npy
data_out/params.npy
data_out/splits.npz (after next step)
metadata .npz + logs .json

# Create fixed splits
python -m data.make_splits --data-out data_out --seed 42 --overwrite

# Visualize tokenization
python -m eval.qc_tokens --data-out data_out --out plots/qc_tokens --n 64

# Train inverse models (Transformer + BiLSTM)
mkdir -p models_out

Hybrid (main condition), 3‑seed ensemble:

python -m models.train --run-name cmc --features hybrid --train-seed 2 --sp

  --h5 data/synthetic_cmc_dataset.h5 \
  --data-out data_out \
  --split-seed 42 \
  --outdir results/neurips_sbi \
  --device cpu \
  --n-post-samples 200
